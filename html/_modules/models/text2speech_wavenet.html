[docs]def save_audio(signal, logdir, step, sampling_rate, mode):
  signal = np.float32(signal)
  file_name = '{}/sample_step{}_{}.wav'.format(logdir, step, mode)
  if logdir[0] != '/':
    file_name = "./" + file_name
  write(file_name, sampling_rate, signal)
